Dialog Widget

Example
Source Code

jPulse provides several types of dialog widgets:

<!-- HTML Buttons -->
<button id="alertBtn" class="jp-btn jp-btn-danger">Alert Dialog</button>
<button id="infoBtn" class="jp-btn jp-btn-primary">Info Dialog</button>
<button id="confirmBtn" class="jp-btn jp-btn-warning">Confirm Dialog</button>

<!-- JavaScript -->
<script>
// Alert dialog
document.getElementById('alertBtn').addEventListener('click', async () => {
    await jPulse.UI.alertDialog('This is an alert message!');
});

// Info dialog
document.getElementById('infoBtn').addEventListener('click', async () => {
    await jPulse.UI.infoDialog('This is an informational message.');
});

// Confirm dialog
document.getElementById('confirmBtn').addEventListener('click', async () => {
    const result = await jPulse.UI.confirmDialog({
        title: 'Confirm Action',
        message: 'Are you sure you want to proceed?',
        buttons: {
            'Cancel': () => false,
            'Yes, Continue': () => true
        }
    });

    if (result.confirmed) {
        jPulse.showSlideDownSuccess('Action confirmed!');
    }
});

// Custom dialog with HTML content
document.getElementById('customBtn').addEventListener('click', async () => {
    await jPulse.UI.infoDialog(
        '<p>This dialog supports <b>HTML content</b>:</p>' +
        '<ul><li>Lists</li><li>Formatting</li><li>Links</li></ul>',
        {
            title: 'Custom Content',
            minWidth: 400,
            minHeight: 200
        }
    );
});
</script>

Form Validation Helpers

Example
Source Code

jPulse provides form validation helpers for common scenarios:

<!-- HTML Form -->
<form id="demo-form" class="jp-form">
    <div class="jp-form-group">
        <label for="email">Email Address:</label>
        <input type="email" id="email" name="email" class="jp-form-input" required>
        <div class="jp-form-error" id="email-error"></div>
    </div>

    <button type="submit" class="jp-btn jp-btn-primary">Validate Form</button>
</form>

<!-- JavaScript Validation -->
<script>
document.getElementById('demo-form').addEventListener('submit', async (e) => {
    e.preventDefault();

    // Clear previous errors
    jPulse.form.clearErrors('demo-form');

    // Validate form
    const isValid = jPulse.form.validate('demo-form', {
        email: {
            required: true,
            email: true
        },
        password: {
            required: true,
            minLength: 6
        }
    });

    if (isValid) {
        jPulse.showSlideDownSuccess('Form validation passed!');
    } else {
        jPulse.showSlideDownError('Please fix the errors above.');
    }
});
</script>
