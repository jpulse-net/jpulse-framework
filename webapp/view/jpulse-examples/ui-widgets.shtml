Dialog Widget

What you learn: How to create modal dialogs for user interactions including alerts, confirmations, and custom content dialogs with async/await patterns.

Behind the scenes: Uses jPulse's dialog system with promise-based responses, customizable buttons, HTML content support, and automatic focus management.

jPulse provides several types of dialog widgets:

<!-- HTML Buttons --> <button id="alertBtn" class="jp-btn jp-btn-danger">Alert Dialog</button> <button id="infoBtn" class="jp-btn jp-btn-primary">Info Dialog</button> <button id="successDialogBtn" class="jp-btn jp-btn-success">Success Dialog</button> <button id="confirmBtn" class="jp-btn jp-btn-warning">Confirm Dialog</button> <button id="complexBtn" class="jp-btn jp-btn-secondary">Complex Dialog</button> <!-- JavaScript --> <script> // Alert dialog - simple usage document.getElementById('alertBtn').addEventListener('click', async () => { await jPulse.UI.alertDialog('This is an alert dialog with important information!'); }); // Info dialog - with title as 2nd parameter document.getElementById('infoBtn').addEventListener('click', async () => { await jPulse.UI.infoDialog('This is an informational dialog with helpful details.', 'Important Notice'); }); // Success dialog - new in v1.0.4 document.getElementById('successDialogBtn').addEventListener('click', async () => { await jPulse.UI.successDialog('Operation completed successfully!', 'Success'); }); // Confirm dialog document.getElementById('confirmBtn').addEventListener('click', async () => { const result = await jPulse.UI.confirmDialog({ title: 'Confirm Action', message: 'Are you sure you want to proceed?', buttons: [ 'Cancel', 'Yes, Continue' ], // Alternative syntax: //buttons: { // 'Cancel': () => {}, // Don't return anything, let dialog close // 'Yes, Continue': () => {} // Don't return anything, let dialog close //} }); if (result.button === 'Yes, Continue') { jPulse.UI.toast.success('Action confirmed!'); } else { jPulse.UI.toast.info('Action cancelled.'); } }); // Custom dialog with HTML content - using options object document.getElementById('customBtn').addEventListener('click', async () => { await jPulse.UI.infoDialog( '<p>This dialog supports <b>HTML content</b>:</p>' + '<ul><li>Lists</li><li>Formatting</li><li>Links</li></ul>', { title: 'Custom Content', minWidth: 400, minHeight: 200 } ); }); // Complex dialog with nested dialogs and form elements document.getElementById('complexBtn').addEventListener('click', async () => { const fileListHtml = ` <p>Select files to delete:</p> <div style="text-align: left; margin: 15px 0; line-height: 1.8;"> <div><input type="checkbox" id="file1" checked> document1.pdf</div> <div><input type="checkbox" id="file2" checked> image.jpg</div> <div><input type="checkbox" id="file3"> backup.zip</div> <div><input type="checkbox" id="file4" checked> notes.txt</div> </div> `; const result = await jPulse.UI.confirmDialog({ title: 'Delete Files', message: fileListHtml, buttons: { 'Cancel': () => false, 'Delete Selected Files': async (dialog) => { const checkboxes = dialog.querySelectorAll('input:checked'); const files = Array.from(checkboxes).map(cb => cb.nextSibling.textContent.trim()); if (files.length === 0) { await jPulse.UI.alertDialog('Please select at least one file.'); return true; // Keep dialog open } const confirmResult = await jPulse.UI.confirmDialog({ title: 'Confirm Deletion', message: `Delete ${files.length} file(s)?`, buttons: { 'Cancel': () => false, 'Yes, Delete': () => true } }); if (confirmResult.confirmed) { jPulse.UI.toast.success(`Deleted ${files.length} file(s)!`); return false; // Close both dialogs } return true; // Keep first dialog open } } }); }); </script>
